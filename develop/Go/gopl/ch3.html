func z4(z complex128) color.Color {
	// f(z) = z^4 -1
	f := func(z complex128) complex128 {
		return z*z*z*z - 1
	}
	// f(z)/f'(z) = (z^4 -1)/(4z^3)
	fPrime := func(z complex128) complex128 {
		return (z - 1/(z*z*z)) / 4
	}
	return newton(z, f, fPrime)
}
